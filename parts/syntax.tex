\subsection{Literals}


\begin{figure}
	\begin{grammar}
		<int> ::= ('0'-'9')+
		
		<hex> ::= ('0'-'0'|'A'-'F'|'a'-'f')+
		
		<Integer> ::= '-'? <int> ('i' <int>?)?
		
		<HexInteger> ::= "0x" <hex> ('i' <int>?)?
		
		<long> ::= '-'? <int> 'L'
		
		<Float> ::= '-'? <int>? '.' <int> (('e'|'E') '-'? <int>)? ('f' <int>?)?
		
		<string> ::= '\verb|"|' \textasciitilde['\verb|"|']* '\verb|"|'
		
		<Identifier> ::= '\verb|`|' \textasciitilde['\verb|`|']+ '\verb|`|'
	\end{grammar}

\end{figure}
